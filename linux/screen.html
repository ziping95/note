# Screen

# 关闭一个已经detached会话

screen -X -S seesion-id quit

# 创建一个会话

screen -S yourname

# 查看当前会话列表

screen -ls

# 回到这个指定的会话中

screen -r yourname

# 远程detach某个session

screen -d yourname

# 结束当前session并回到yourname这个session

screen -d -r yourname
